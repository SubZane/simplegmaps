Open address links in native map apps

  • Google Maps (for smartphone)
  • Apple Maps (OSX, iOS)
  • Maps on WindowsPhone7
  • Google Maps (Normal browser link)
  • Native Map App Link (Intelligent link)

javascript


jQuery(document).ready(function ($) {
  $('#simplemap').simplegmaps();

  $('.getGoogleMapLink').attr('href', $('#simplemap-1').simplegmaps('getGoogleMapLink', 'Remi 145 W 53rd St, New York, NY, United States'));
  $('.getAppleMapsLink').attr('href', $('#simplemap-1').simplegmaps('getAppleMapsLink', 'Remi 145 W 53rd St, New York, NY, United States'));
  $('.getWindowsPhone7MapLink').attr('href', $('#simplemap-1').simplegmaps('getWindowsPhone7MapLink', 'Remi 145 W 53rd St, New York, NY, United States'));
  $('.getDesktopMapLink').attr('href', $('#simplemap-1').simplegmaps('getDesktopMapLink', 'Remi 145 W 53rd St, New York, NY, United States'));
  $('.getNativeMapLink').attr('href', $('#simplemap-1').simplegmaps('getNativeMapLink', 'Remi 145 W 53rd St, New York, NY, United States'));
});
